```typescript
export interface UserProfile {
  level: 'beginner' | 'intermediate' | 'advanced';
  style: 'control' | 'power' | 'balanced';
  budget_max: number;
}

export interface ChatRequest {
  message: string;
  skill_level: 'beginner' | 'intermediate' | 'advanced';
  budget_brl: number;
  style?: string;
}

export interface ChatRecommendation {
  paddle_id: number;
  name: string;
  brand: string;
  price_min_brl: number;
  affiliate_url: string;
  similarity_score: number;
}
```

<!-- From backend: FastAPI SSE event format -->
FastAPI POST /chat SSE events:
```
event: recommendations\ndata: {"paddles": [ChatRecommendation...]}\n\n
event: reasoning\ndata: {"text": "..."}\n\n
event: degraded\ndata: {"paddles": [...]}\n\n
event: done\ndata: {"tokens": N, "latency_ms": N, "model": "...", "cache_hit": bool}\n\n
event: error\ndata: {"error": "..."}\n\n
```

Vercel AI SDK data stream format (target):
```
0:"token text chunk"     // text token
2:[{paddle_data}]        // data array (annotations)
d:{"finishReason":"stop"} // done signal
e:{"error":"..."}        // error
```
