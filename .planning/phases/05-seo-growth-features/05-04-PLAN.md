       export function markAffiliateContent(content: string, brandName: string) {
         // Helper to inject FTC disclosure before first Mercado Livre link
         const mercadoLivreRegex = /https:\/\/mercadolivre\.com/;
         if (mercadoLivreRegex.test(content)) {
           // Inject disclosure marker; JSX rendering handles replacement
           return content.replace(
             mercadoLivreRegex,
             '[FTC_DISCLOSURE]$&'
           );
         }
         return content;
       }

       export function generateBlogMetadata(title: string, description: string, slug: string) {
