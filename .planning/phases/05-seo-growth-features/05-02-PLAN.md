       import { Metadata } from 'next';

       const FASTAPI_URL = process.env.NEXT_PUBLIC_FASTAPI_URL || 'http://localhost:8000';

       export async function fetchProductData(brand: string, modelSlug: string) {
         const res = await fetch(`${FASTAPI_URL}/paddles?brand=${brand}&model_slug=${modelSlug}`);
         if (!res.ok) throw new Error(`Product not found: ${brand}/${modelSlug}`);
         const data = await res.json();
         return data.paddles?.[0] || null;
       }

       export async function generateProductMetadata(
         brand: string,
         modelSlug: string,
         paddle: any
       ): Promise<Metadata> {
         return {
           title: `${paddle.name} - PickleIQ`,

           openGraph: {
             type: 'website',
             url: `https://pickleiq.com/paddles/${brand}/${modelSlug}`,

       import { Paddle } from '@/types/paddle';

       export function ProductSchema({ paddle, url }: { paddle: Paddle; url: string }) {
         const schema = {
           '@context': 'https://schema.org/',
           '@type': 'Product',
           name: paddle.name,
           brand: { '@type': 'Brand', name: paddle.brand },
           description: paddle.description,
           image: paddle.image_url,
           offers: {
             '@type': 'AggregateOffer',
             priceCurrency: 'BRL',
             price: paddle.price_brl.toString(),
             availability: paddle.in_stock ? 'https://schema.org/InStock' : 'https://schema.org/OutOfStock',
           },
           aggregateRating: {
             '@type': 'AggregateRating',
             ratingValue: paddle.rating.toString(),
             reviewCount: paddle.review_count.toString(),
           },
           url: url,
         };

         return (
           <script type="application/ld+json" dangerouslySetInnerHTML={{ __html: JSON.stringify(schema) }} />
         );
       }
       ```

       import { ProductSchema } from '@/components/schema/product-schema';
       import dynamic from 'next/dynamic';

       const PriceHistoryChart = dynamic(() => import('@/components/price-history-chart'), { ssr: false });

       export async function generateMetadata({ params }: { params: { brand: string; model-slug: string } }) {
         const paddle = await fetchProductData(params.brand, params['model-slug']);
         if (!paddle) return { title: 'Product not found' };
         return generateProductMetadata(params.brand, params['model-slug'], paddle);
       }

       export const revalidate = false; // SSR: always fresh

       export default async function ProductPage({ params }: { params: { brand: string; model-slug: string } }) {
         const paddle = await fetchProductData(params.brand, params['model-slug']);
         if (!paddle) notFound();

         const canonicalUrl = `https://pickleiq.com/paddles/${params.brand}/${params['model-slug']}`;

         return (
           <>
             <ProductSchema paddle={paddle} url={canonicalUrl} />
             <article>
               <h1>{paddle.name}</h1>
               <p>{paddle.description}</p>
               <div>Price: R$ {paddle.price_brl.toFixed(2)}</div>
               <PriceHistoryChart paddleId={paddle.id} days={90} />
             </article>
           </>
         );
       }

    2. Create `lib/revalidate.ts`:

       import { revalidateTag, revalidatePath } from 'next/cache';

       export async function revalidatePaddlePages(paddleId?: number) {
         if (paddleId) {
           // On-demand revalidation for single paddle
           revalidateTag(`paddle:${paddleId}`);
         } else {
           // Bulk revalidation (used by scheduled workers)
           revalidatePath('/paddles');
         }
       }

       export async function revalidateWebhook(req: Request) {
         const auth = req.headers.get('authorization');
         if (auth !== `Bearer ${process.env.REVALIDATE_SECRET}`) {
           return new Response(JSON.stringify({ error: 'Unauthorized' }), { status: 401 });
         }

         const { paddle_id } = await req.json();
         revalidatePaddlePages(paddle_id);
         return new Response(JSON.stringify({ revalidated: true }));
       }
       ```
