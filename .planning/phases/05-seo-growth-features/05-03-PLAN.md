```sql
CREATE MATERIALIZED VIEW latest_prices AS
  SELECT DISTINCT ON (paddle_id, retailer_id)
```

       ```python
       from fastapi import APIRouter, Query
       from sqlalchemy import func
       from datetime import datetime, timedelta
       from backend.db import get_connection

       router = APIRouter()

       @router.get("/paddles/{paddle_id}/price-history")
       async def get_price_history(paddle_id: int, days: int = Query(90, ge=1, le=180)):
           async with get_connection() as conn:
               cutoff_date = datetime.now() - timedelta(days=days)

               # Fetch all snapshots for this paddle in date range
               rows = await conn.fetch("""
                   SELECT
                     ps.retailer_id,
                     ps.price_brl,
                     ps.scraped_at::date AS date,
                     r.name AS retailer_name
                   FROM price_snapshots ps
                   JOIN retailers r ON r.id = ps.retailer_id
                   WHERE ps.paddle_id = $1 AND ps.scraped_at >= $2
                   ORDER BY ps.retailer_id, ps.scraped_at ASC
               """, paddle_id, cutoff_date)

               # Group by retailer and calculate percentiles
               retailer_data = {}
               for row in rows:
                   retailer = row['retailer_name']
                   if retailer not in retailer_data:
                       retailer_data[retailer] = []
                   retailer_data[retailer].append({
                       'date': row['date'].isoformat(),
                       'price': float(row['price_brl']),
                   })

               # Calculate P20 per retailer
               result = []
               for retailer, prices in retailer_data.items():
                   prices_sorted = sorted([p['price'] for p in prices])
                   p20_idx = int(len(prices_sorted) * 0.2)
                   p20 = prices_sorted[p20_idx] if p20_idx < len(prices_sorted) else prices_sorted[-1]

                   for price_point in prices:
                       result.append({
                           'retailer': retailer,
                           'date': price_point['date'],
                           'price': price_point['price'],
                           'p20': p20,
                           'is_good_time': price_point['price'] <= p20,
                       })

               return result
       ```

    2. Create `lib/price-history.ts`:
       ```typescript
       export interface PriceHistoryPoint {
         retailer: string;
         date: string;
         price: number;
         p20: number;
         is_good_time: boolean;
       }

       export function percentile20(prices: number[]): number {
         const sorted = [...prices].sort((a, b) => a - b);
         const idx = Math.floor(sorted.length * 0.2);
         return sorted[idx] || sorted[sorted.length - 1];
       }

       export function isGoodTimeToBuy(price: number, p20: number): boolean {
         return price <= p20;
       }

       export async function getPriceHistory(
         paddleId: number,
         days: number = 90
       ): Promise<PriceHistoryPoint[]> {
         const res = await fetch(`/api/paddles/${paddleId}/price-history?days=${days}`);
         if (!res.ok) return [];
         return res.json();
       }
       ```

       ```typescript
       'use client';

       import { useEffect, useState } from 'react';
       import { LineChart, Line, CartesianGrid, Tooltip, Legend, XAxis, YAxis, ResponsiveContainer } from 'recharts';
       import { getPriceHistory, PriceHistoryPoint } from '@/lib/price-history';

       interface PriceHistoryChartProps {
         paddleId: number;
         days?: number;
       }

       export default function PriceHistoryChart({ paddleId, days = 90 }: PriceHistoryChartProps) {
         const [data, setData] = useState<PriceHistoryPoint[]>([]);
         const [loading, setLoading] = useState(true);

         useEffect(() => {
           getPriceHistory(paddleId, days)
             .then((result) => {
               // Transform for Recharts: group by date, spread retailers as separate fields
               const grouped = result.reduce((acc, point) => {
                 const existing = acc.find((x) => x.date === point.date);
                 if (existing) {
                   existing[`${point.retailer}_price`] = point.price;
                   existing[`${point.retailer}_is_good`] = point.is_good_time;
                 } else {
                   acc.push({
                     date: point.date,
                     [`${point.retailer}_price`]: point.price,
                     [`${point.retailer}_is_good`]: point.is_good_time,
                   });
                 }
                 return acc;
               }, [] as any[]);

               setData(grouped.sort((a, b) => new Date(a.date).getTime() - new Date(b.date).getTime()));
             })
             .finally(() => setLoading(false));
         }, [paddleId, days]);

         if (loading) return <div>Carregando histórico de preços...</div>;
         if (data.length === 0) return <div>Sem dados de preço disponíveis.</div>;

         const retailerNames = Array.from(
           new Set(data.flatMap((d) => Object.keys(d).filter((k) => k.endsWith('_price')).map((k) => k.replace('_price', ''))))
         );

         return (
           <ResponsiveContainer width="100%" height={400}>
             <LineChart data={data}>
               <CartesianGrid strokeDasharray="3 3" />
               <XAxis dataKey="date" />
               <YAxis label={{ value: 'Preço (R$)', angle: -90, position: 'insideLeft' }} />
               <Tooltip
                 formatter={(value) => `R$ ${(value as number).toFixed(2)}`}
                 labelFormatter={(label) => `Data: ${label}`}
               />
               <Legend />
               {retailerNames.map((retailer, idx) => (
                 <Line
                   key={`${retailer}_price`}
                   type="monotone"
                   dataKey={`${retailer}_price`}
                   stroke={['#3b82f6', '#ef4444', '#10b981'][idx % 3]}
                   name={`${retailer}`}
                   connectNulls
                 />
               ))}
             </LineChart>
           </ResponsiveContainer>
         );
       }
       ```

    2. Create `backend/workers/price_alert_check.py`:
       ```python
       import os
       import asyncio
       import httpx
       from datetime import datetime
       from psycopg import AsyncConnection

       DATABASE_URL = os.getenv('DATABASE_URL')
       RESEND_API_KEY = os.getenv('RESEND_API_KEY')
       FASTAPI_URL = os.getenv('FASTAPI_URL', 'http://localhost:8000')

       async def check_price_alerts():
           async with await AsyncConnection.connect(DATABASE_URL) as conn:
               # Fetch active alerts
               alerts = await conn.execute("""
                   SELECT pa.id, pa.user_id, pa.paddle_id, pa.price_target, u.email
                   FROM price_alerts pa
                   JOIN users u ON u.id = pa.user_id
                   WHERE pa.active = true
               """)

               triggered_count = 0
               for alert_id, user_id, paddle_id, price_target, email in alerts:
                   # Get current price from latest_prices view
                   latest = await conn.execute("""
                       SELECT MIN(price_brl) as current_price
                       FROM latest_prices
                       WHERE paddle_id = $1
                   """, paddle_id)

                   current_price = latest[0][0] if latest else None
                   if current_price is None:
                       print(f"[SKIP] Alert {alert_id}: no price data for paddle {paddle_id}")
                       continue

                   # Check if threshold breached
                   if current_price <= price_target:
                       # Send email via Resend
                       async with httpx.AsyncClient() as client:
                           res = await client.post(
                               'https://api.resend.com/emails',
                               headers={'Authorization': f'Bearer {RESEND_API_KEY}'},
                               json={
                                   'from': 'alerts@pickleiq.com',
                                   'to': email,
                                   'subject': f'Alerta de preço ativado! Raquete agora R$ {current_price:.2f}',
                                   'html': f'<p>Ótima notícia! Seu alerta de preço foi ativado.</p><p>Preço atual: R$ {current_price:.2f}</p>',
                               },
                           )

                           if res.status_code == 200:
                               await conn.execute(
                                   "UPDATE price_alerts SET last_triggered = $1 WHERE id = $2",
                                   datetime.now(),
                                   alert_id,
                               )
                               triggered_count += 1
                               print(f"[OK] Alert {alert_id}: email sent to {email}")
                           else:
                               print(f"[ERROR] Alert {alert_id}: Resend API error {res.status_code}")

               print(f"\n[SUMMARY] Triggered {triggered_count} alerts")

       if __name__ == '__main__':
           asyncio.run(check_price_alerts())
       ```
