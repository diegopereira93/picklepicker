    ```typescript
    const productSchema = {
      '@context': 'https://schema.org',
      '@type': 'Product',
      name: paddle.name,
      description: paddle.description || `${paddle.brand} ${paddle.name} — raquete de pickleball`,

        url: `https://pickleiq.com/catalog/${paddle.model_slug || paddle.id}`,
        seller: {
          '@type': 'Organization',
          name: 'PickleIQ',
        },
      },
      ...(paddle.rating && {
        aggregateRating: {
          '@type': 'AggregateRating',
          ratingValue: paddle.rating,
          reviewCount: paddle.review_count || 1,
          bestRating: 5,
          worstRating: 1,
        },
      }),
    }
    ```

    ```tsx
    <script
      type="application/ld+json"
      dangerouslySetInnerHTML={{ __html: JSON.stringify(productSchema) }}
    />
    ```

    Note: `JSON.stringify` is safe here — no user-controlled data is injected as HTML. The `dangerouslySetInnerHTML` is the standard React pattern for JSON-LD and is safe because the content is JSON, not executable HTML.
