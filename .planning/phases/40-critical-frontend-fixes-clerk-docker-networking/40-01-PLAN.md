```typescript
export function ClerkWrapper({ children }: { children: ReactNode }) {
  const [mounted, setMounted] = useState(false);
```

  <action>
Split ClerkAuthButtons and MobileClerkAuth into wrapper + inner components to fix the Rules of Hooks violation.

The root cause: `useAuth()` is a Clerk hook that throws if called outside ClerkProvider. The current code calls `useClerkAvailable()` then conditionally returns null, then calls `useAuth()`. This violates React's Rules of Hooks because the hook order changes between renders (1 hook vs 2 hooks).

Fix pattern: Create inner components that ONLY mount when clerk is available, so `useAuth()` is always called within ClerkProvider:

```typescript
// NEW: Inner component that only mounts inside ClerkProvider
function ClerkAuthButtonsInner() {
  const { isSignedIn } = useAuth()  // Safe: always inside ClerkProvider
  return (
    <>
      {!isSignedIn && (
        <SignInButton mode="modal">
          <Button variant="outline" size="sm">Entrar</Button>
        </SignInButton>
      )}
      {isSignedIn && <UserButton afterSignOutUrl="/" />}
    </>
  )
}

// Wrapper: always calls same hooks (just useClerkAvailable)
export function ClerkAuthButtons() {
  const clerkAvailable = useClerkAvailable()
  if (!clerkAvailable) return null
  return <ClerkAuthButtonsInner />
}
```

Apply the same pattern to MobileClerkAuth → MobileClerkAuthInner.

Keep all imports from `@clerk/nextjs` (SignInButton, UserButton, useAuth) and `@/components/clerk-provider` (useClerkAvailable).
Keep "use client" directive.

```typescript
import { describe, it, expect, vi } from 'vitest'

// Mock @clerk/nextjs — useAuth throws if called outside ClerkProvider
vi.mock('@clerk/nextjs', () => ({
  useAuth: vi.fn(),
  SignInButton: ({ children }: { children: React.ReactNode }) => children,
  UserButton: () => null,
}))

// Mock the useClerkAvailable hook
vi.mock('@/components/clerk-provider', () => ({
  useClerkAvailable: vi.fn(),
}))

import { useAuth } from '@clerk/nextjs'
import { useClerkAvailable } from '@/components/clerk-provider'
import { ClerkAuthButtons, MobileClerkAuth } from '@/components/layout/clerk-auth-buttons'

describe('ClerkAuthButtons', () => {
  it('returns null when clerk is unavailable', () => {
    vi.mocked(useClerkAvailable).mockReturnValue(false)
    const { container } = render(<ClerkAuthButtons />)
    expect(container.innerHTML).toBe('')
    // useAuth must NOT be called when clerk is unavailable
    expect(useAuth).not.toHaveBeenCalled()
  })

  it('renders sign-in button when clerk available and not signed in', () => {
    vi.mocked(useClerkAvailable).mockReturnValue(true)
    vi.mocked(useAuth).mockReturnValue({ isSignedIn: false } as any)
    render(<ClerkAuthButtons />)
    expect(screen.getByText('Entrar')).toBeInTheDocument()
  })

  it('renders user button when clerk available and signed in', () => {
    vi.mocked(useClerkAvailable).mockReturnValue(true)
    vi.mocked(useAuth).mockReturnValue({ isSignedIn: true } as any)
    const { container } = render(<ClerkAuthButtons />)
    // UserButton is mocked as null, so no sign-in button should appear
    expect(screen.queryByText('Entrar')).not.toBeInTheDocument()
  })
})

describe('MobileClerkAuth', () => {
  it('returns null when clerk is unavailable', () => {
    vi.mocked(useClerkAvailable).mockReturnValue(false)
    const { container } = render(<MobileClerkAuth />)
    expect(container.innerHTML).toBe('')
    expect(useAuth).not.toHaveBeenCalled()
  })
})
```
