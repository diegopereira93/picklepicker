export function resolveFastApiUrl(env: Record<string, string | undefined>): string {
  return env.FASTAPI_INTERNAL_URL || env.FASTAPI_URL || 'http://localhost:8000'

Test file structure:
```typescript
import { describe, it, expect } from 'vitest'
import { resolveFastApiUrl } from '@/app/api/chat/route'

describe('resolveFastApiUrl', () => {
  it('prefers FASTAPI_INTERNAL_URL (Docker)', () => {
    expect(resolveFastApiUrl({
      FASTAPI_INTERNAL_URL: 'http://backend:8000',
      FASTAPI_URL: 'http://localhost:8000',
    })).toBe('http://backend:8000')
  })

  it('falls back to FASTAPI_URL when INTERNAL not set', () => {
    expect(resolveFastApiUrl({
      FASTAPI_URL: 'http://localhost:8000',
    })).toBe('http://localhost:8000')
  })

  it('defaults to localhost:8000 when neither set', () => {
    expect(resolveFastApiUrl({})).toBe('http://localhost:8000')
  })
})
```
